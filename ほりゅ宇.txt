#include <bits/stdc++.h>
using namespace std;
 
int main() {
  int N;
  cin >> N;
  string vec[N];
  
  for (int i = 0; i < N; i++) {
    cin >> vec[i];
  }
  
  sort(vec,vec+N);
  int ans = 1; 

  for (int i = 0; i < N-1; i++) {
    if(vec[i] != vec[i+1]){
        ans++;
    }
  }
  
  cout << ans <<endl;
}
